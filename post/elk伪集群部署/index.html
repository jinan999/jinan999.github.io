ELK伪集群部署

Jan 30, 2023

环境：此测试是在一台主机上部署 ELK 伪集群，主机IP：192.168.0.129 修改服务器内存参数(必须修改，不然elk启动后会报错)： sysctl -w vm.max_map_count=262144 1.部署 elasticsearch 节点1：es01 #创建文件 mkdir -p /usr/local/elk01 touch /usr/local/elk01/docker-compose.yml /usr/local/elk01/es01.yml

#编写docker-compose.yml cat > /usr/local/elk01/docker-compose.yml «EOF version: ‘2.2’ services: es01: image: docker.elastic.co/elasticsearch/elasticsearch:7.9.3 container_name: es01 restart: always environment: - bootstrap.memory_lock=true - “ES_JAVA_OPTS=-Xms2g -Xmx2g” ulimits: memlock: soft: -1 hard: -1 volumes: - data01:/usr/share/elasticsearch/data - /elk_backups/data:/usr/share/elasticsearch/backup - /usr/local/elk01/es01.yml:/usr/share/elasticsearch/config/elasticsearch.yml:ro ports: - 9292:9200 - 9300:9300 networks: - elastic volumes: data01: driver: local networks: elastic: driver: bridge EOF

#编写配置文件 es01.yml cat > /usr/local/elk01/es01.yml «EOF path.repo: ["/usr/share/elasticsearch/backup"] network.host: 0.0.0.0 http.cors.enabled: true http.cors.allow-origin: “*” node.name: es01 cluster.name: es-cluster-prod node.master: true gateway.expected_nodes: 1 discovery.zen.minimum_master_nodes: 1 transport.tcp.port: 9300 indices.fielddata.cache.size: 40% indices.breaker.total.use_real_memory: false network.publish_host: 192.168.0.129 discovery.seed_hosts: 192.168.0.129:9300,192.168.0.129:9301 cluster.initial_master_nodes: 192.168.0.129:9300 EOF

#启动容器 cd /usr/local/elk01 docker-compose up -d 节点2：es02 #创建文件 mkdir -p /usr/local/elk02 touch /usr/local/elk02/docker-compose.yml /usr/local/elk02/es02.yml

#编写docker-compose.yml cat > /usr/local/elk02/docker-compose.yml «EOF version: ‘2.2’ services: es02: image: docker.elastic.co/elasticsearch/elasticsearch:7.9.3 container_name: es02 restart: always environment: - bootstrap.memory_lock=true - “ES_JAVA_OPTS=-Xms2g -Xmx2g” ulimits: memlock: soft: -1 hard: -1 volumes: - data02:/usr/share/elasticsearch/data - /elk_backups/data:/usr/share/elasticsearch/backup - /usr/local/elk02/es02.yml:/usr/share/elasticsearch/config/elasticsearch.yml:ro ports: - 9301:9301 networks: - elastic volumes: data02: driver: local networks: elastic: driver: bridge EOF

#编写配置文件 es02.yml cat > /usr/local/elk02/es02.yml «EOF path.repo: ["/usr/share/elasticsearch/backup"] network.host: 0.0.0.0 http.cors.enabled: true http.cors.allow-origin: “*” node.name: es02 cluster.name: es-cluster-prod node.master: false gateway.expected_nodes: 1 discovery.zen.minimum_master_nodes: 1 transport.tcp.port: 9301 indices.fielddata.cache.size: 40% indices.breaker.total.use_real_memory: false network.publish_host: 192.168.0.129 discovery.seed_hosts: 192.168.0.129:9300 ,192.168.0.129:9301 cluster.initial_master_nodes: 192.168.0.129:9300 EOF

#启动容器 cd /usr/local/elk02 docker-compose up -d 2.部署 kibana #创建文件 mkdir -p /usr/local/kibana touch /usr/local/kibana/kibana.yml

#编写配置文件 kibana.yml cat > /usr/local/kibana/kibana.yml «EOF server.name: kibana server.host: “0.0.0.0” server.maxPayloadBytes: 10485760 #传入服务器请求的最大有效负载大小(以字节为单位) elasticsearch.hosts: [ “http://192.168.0.129:9292”,“http://192.168.0.129:9293” ] xpack.monitoring.ui.container.elasticsearch.enabled: true xpack.reporting.csv.maxSizeBytes: 2097152000 # csv文件大小,默认为10485760（10MB） xpack.reporting.queue.timeout: 3000000 # 超时时间,默认是120000(2分钟) EOF

#启动 kibana docker run –name kibana –privileged=true –restart=always -d -e “I18N_LOCALE=zh-CN” -p 5692:5601 -v /usr/local/kibana/kibana.yml:/usr/share/kibana/config/kibana.yml -v /etc/localtime:/etc/localtime:ro docker.elastic.co/kibana/kibana:7.9.3

3.部署 filebeat #创建文件 mkdir /filebeat_data

#编写filebeat配置文件 cat > /filebeat_data/filebeat.yml «EOF filebeat.inputs:

  • type: log paths:
    • /var/log/gamelog//ActionLogs/.log json.keys_under_root: true json.overwrite_keys: true tags: [“prod_logs”] harvester_buffer_size: 1638400 scan_frequency: 5m http.enabled: true http.port: 5067 setup.template.name: "" setup.template.pattern: "" setup.template.enabled: false setup.template.overwrite: true processors:
  • timestamp: field: time timezone: Asia/Shanghai layouts: - ‘2006-01-02 15:04:05’ - ‘2006-01-02 15:04:05.999’ test: - ‘2019-06-22 16:33:51’
  • truncate_fields: fields: - time max_characters: 7 fail_on_error: false ignore_missing: true
  • drop_fields: fields: [“host”,“agent”,“esc”,“input”] output.elasticsearch: hosts: [“192.168.0.129:9292”] bulk_max_size: 500 indices:
    • index: “%{[logEvt]}-%{[time]}” when.contains: tags: “prod_logs” EOF

#下载镜像 docker pull elastic/filebeat:7.9.3

#启动 filebeat docker run -u root -d –restart=always –name=filebeat -v /data/game_server/target/Logs:/var/log/gamelog/ -v /filebeat_data/filebeat.yml:/usr/share/filebeat/filebeat.yml -v /filebeat_data/registry/:/usr/share/filebeat/data/registry/ elastic/filebeat:7.9.3
